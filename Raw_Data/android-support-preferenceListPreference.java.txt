    public int findIndexOfValue(@Nullable String value) {
        final CharSequence[] entryValues = getEntryValues();
        if (value != null && entryValues != null) {
            for (int i = entryValues.length - 1; i >= 0; --i) {
                if (value.contentEquals(entryValues[i])) {
                    return i;
                }
            }
        }
        return -1;
    }
